  window.requestFileSystem(LocalFileSystem.PERSISTENT, 0, function (fs) {
    fs.root.getFile(submitFileName, {}, function (fileEntry) {
      fileEntry.file(function (file) {
        var reader = new FileReader();
        reader.onloadend = function () {
          console.log("File content:", this.result);
          try {
            const data = JSON.parse(this.result);
            let text = data.map(d => JSON.stringify(d)).join('\n\n');
            app.dialog.alert(text || "No data.");
          } catch (e) {
            app.dialog.alert("Invalid JSON.");
          }
        };
        reader.readAsText(file);
      });
    });
  }, onError);
}

function showSubmissions() {
  const filePath = cordova.file.externalRootDirectory + "Download/" + submitFileName;

  window.resolveLocalFileSystemURL(filePath, function (fileEntry) {
    fileEntry.file(function (file) {
      const reader = new FileReader();
      reader.onloadend = function () {
        try {
          const data = JSON.parse(this.result);
          app.dialog.alert(
            'Файлът е записан в: <br><code>' + filePath + '</code><br><br>Съдържание:<br>' +
            '<pre>' + JSON.stringify(data, null, 2) + '</pre>'
          );
        } catch (e) {
          app.dialog.alert("Неуспешно четене на съдържанието.");
        }
      };
      reader.readAsText(file);
    }, onError);
  }, function (err) {
    console.error("File not found:", err);
    app.dialog.alert("Файлът не е намерен в:<br><code>" + filePath + "</code>");
  });
}
